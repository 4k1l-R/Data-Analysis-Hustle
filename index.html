library(ggplot2)

concrete <- concrete[order(concrete$month), ]

m      <- nrow(concrete)
months <- concrete$month
cs_n   <- cumsum(concrete$n)
cs_h   <- cumsum(concrete$h)
tot_n  <- sum(concrete$n)
tot_h  <- sum(concrete$h)

# Run prop.test at each cut
pvals <- numeric(m - 1); bp <- numeric(m - 1); ap <- numeric(m - 1)
for (i in 1:(m - 1)) {
  n_b <- cs_n[i]; h_b <- cs_h[i]
  n_a <- tot_n - n_b; h_a <- tot_h - h_b
  bp[i] <- h_b / n_b; ap[i] <- h_a / n_a
  pvals[i] <- prop.test(c(h_b, h_a), c(n_b, n_a), correct = FALSE)$p.value
}

# Results data frame for plotting
res <- data.frame(
  cut_after_month = months[1:(m - 1)],
  before_p = bp, after_p = ap,
  p_value = pvals,
  neglogp = -log(pvals)
)
